#include <stdio.h>
#include <stdlib.h>
#include <math.h>

const int maxn = 5000;

int n, sz;
int x[maxn], y[maxn];

struct elem {
    int a, b, c;
} v[maxn * maxn / 2];

int random(int size) {
    return rand() % size;
}

void sort(int lt, int rt) {
    int i = lt, j = rt;
    int mid = v[lt + random(rt - lt + 1)].c;
    while (i <= j) {
        for (; v[i].c < mid; ++i);
        for (; mid < v[j].c; --j);
        if (i <= j) {
            struct elem tmp = v[i];
            v[i] = v[j];
            v[j] = tmp;
            ++i;
            --j;
        }
    }
    if (lt < j) sort(lt, j);
    if (i < rt) sort(i, rt);
}

int par[maxn];

void unite(int a, int b) {
    if (rand() & 1) {
        int t = a;
        a = b;
        b = t;
    }
    par[a] = b;
}

int get(int v) {
    return v == par[v] ? v : (par[v] = get(par[v]));
}

int main() {
    freopen("unionday.in", "r", stdin);
    freopen("unionday.out", "w", stdout);

    scanf("%d", &n);
    
    for (int i = 0; i < n; ++i) {
        scanf("%d %d", &x[i], &y[i]);
    }

    sz = 0;
    for (int i = 0; i < n; ++i)
        for (int j = i + 1; j < n; ++j) {
            v[sz].a = i;
            v[sz].b = j;
            v[sz].c = (x[i] - x[j]) * (x[i] - x[j]) + (y[i] - y[j]) * (y[i] - y[j]);
            ++sz;
        }

    sort(0, sz - 1);
    
    for (int i = 0; i < n; ++i) par[i] = i;
    double ans = 0;

    for (int i = 0, f, t; i < sz; ++i) {
        f = get(v[i].a);
        t = get(v[i].b);
        if (f != t) {
            unite(f, t);
            ans += sqrt((double)v[i].c);
        }
    }

    printf("%.12lf", ans);
    return 0;
}
